Git 常用命令

git remote add [库别名] [库地址]
git remote rm [库别名]
git push [库别名] [分支]
git status
git cache -rm

#分支管理
git branch 查看所有分支
git branch dev 创建分支
git checkout -b develop master 创建+切换分支
git checkout master 切换到Master分支
git checkout -filename.txt 丢弃工作区修改( - 很重要)
git banch -d bug 删除分支
git merge –no-ff develop 将develop分支合并到当前分支git

#工作区
git stash 挂起工作区.
git stash list 查看挂起的工作区
git stash pop 回复工作区内容并删除记录

#版本回退
git reset -hard HEAD^ 上一个版本
git reset -hard HEAD^^ 上两个版本
git reset -hard HEAD~100 上100个版本
git reset -hard [版本号] 指定版本号

#bug修复的流程
git checkout -b [分支名] //在需要开分支的分支上
…修改,提交git
切换回主分支
git merge –no-ff [分支名]
